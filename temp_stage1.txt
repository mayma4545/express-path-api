  <script>
    function switchTab(name) {
      document.querySelectorAll('.tab-btn').forEach(btn => {
        btn.classList.remove('active', 'text-brand', 'border-brand');
        btn.classList.add('text-gray-500', 'border-transparent');
      });
      document.querySelectorAll('.tab-content').forEach(content => {
        content.classList.remove('active');
        content.classList.add('hidden'); // Also hide the generic ones just in case
      });
      
      const activeBtn = document.getElementById(`tab-${name}`);
      if(activeBtn) {
        activeBtn.classList.remove('text-gray-500', 'border-transparent');
        activeBtn.classList.add('active', 'text-brand', 'border-brand');
      }
      
      const activeContent = document.getElementById(`tab-${name}-content`);
      if(activeContent) {
        activeContent.classList.add('active');
        activeContent.classList.remove('hidden');
      }

      if (name === 'attendees') fetchAttendees('<%= event.id %>');
      if (name === 'analytics') fetchAnalytics('<%= event.id %>');
    }

    // Initialize the default active tab
    switchTab('about');
    
    function toggleBroadcastModal() {
      document.getElementById('broadcastModal').classList.toggle('hidden');
    }

    async function sendBroadcast(eventId) {
        const subject = document.getElementById('bcSubject').value.trim();
        const message = document.getElementById('bcMessage').value.trim();
        
        if (!subject || !message) {
            alert('Please fill out both subject and message.');
            return;
        }
        
        try {
            const res = await fetch(`/organizer/events/${eventId}/broadcast`, {
                method: 'POST',
                headers: {'Content-Type': 'application/json'},
                body: JSON.stringify({ subject, message })
            });
            const data = await res.json();
            if (data.success) {
                alert(data.message);
                toggleBroadcastModal();
                document.getElementById('bcSubject').value = '';
                document.getElementById('bcMessage').value = '';
            } else {
                alert(data.error || data.message || 'Error sending broadcast');
            }
        } catch (error) {
            alert('Network error while sending broadcast');
        }
    }

    async function fetchAttendees(eventId) {
        const container = document.getElementById('attendees-list-container');
        if (!container) return;
        container.innerHTML = '<div class="text-center text-gray-500 py-12 text-sm"><div class="w-8 h-8 border-4 border-brand border-t-transparent rounded-full animate-spin mx-auto mb-4"></div>Loading attendees...</div>';
        
        try {
            const response = await fetch(`/organizer/events/${eventId}/attendees`);
            const data = await response.json();
            
            if (data.attendees && data.attendees.length > 0) {
                container.innerHTML = `
                    <div class="mb-4 text-xs font-bold text-gray-400 uppercase tracking-wider pl-2">${data.attendees.length} Registered Guest(s)</div>
                    <div class="space-y-3">
                        ${data.attendees.map(a => `
                            <div class="flex items-center justify-between bg-white p-4 rounded-2xl shadow-soft border border-gray-50">
                                <div class="flex items-center gap-3">
                                  <div class="w-10 h-10 rounded-full bg-brand-light text-brand flex items-center justify-center font-bold">
                                    ${a.user ? a.user.first_name[0] : '?'}
                                  </div>
                                  <div>
                                      <p class="font-bold text-sm text-textMain">${a.user ? a.user.first_name + ' ' + a.user.last_name : 'Guest User'}</p>
                                      <p class="text-xs text-textMuted">${a.user ? a.user.email : 'No email provided'}</p>
                                  </div>
                                </div>

      async function sendBroadcast(eventId) {
          const subject = document.getElementById('bcSubject').value.trim();
          const message = document.getElementById('bcMessage').value.trim();
          
          if (!subject || !message) {
              alert('Please fill out both subject and message.');
              return;
          }
          
          try {
              const res = await fetch(`/organizer/events/${eventId}/broadcast`, {
                  method: 'POST',
                  headers: {'Content-Type': 'application/json'},
                  body: JSON.stringify({ subject, message })
              });
              const data = await res.json();
              if (data.success) {
                  alert(data.message);
                  toggleBroadcastModal();
                  document.getElementById('bcSubject').value = '';
                  document.getElementById('bcMessage').value = '';
              } else {
                  alert(data.error || data.message || 'Error sending broadcast');
              }
          } catch (error) {
              alert('Network error while sending broadcast');
          }
      }
